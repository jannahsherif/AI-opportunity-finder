async function send() { const input = document.getElementById('user-input'); const query = input.value.trim(); if (!query || loading) return; // UI State: Start Loading loading = true; document.getElementById('status-dot').className = 'sdot loading'; document.getElementById('status-text').innerText = 'searching...'; document.getElementById('empty').style.display = 'none'; // Add User Message to UI const chat = document.getElementById('chat'); chat.innerHTML += `
${query}
`; input.value = ''; try { const res = await fetch('/api/search', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ messages: [{ role: 'user', content: query }], filters: filters, deadline: deadline }) }); const data = await res.json(); // 1. Add AI Response text chat.innerHTML += `
${data.message}
`; // 2. Clear and Show Results const list = document.getElementById('results-list'); list.innerHTML = ''; document.getElementById('results-panel').style.display = 'flex'; document.getElementById('results-count').innerText = `${data.results.length} found`; data.results.forEach(item => { const card = document.createElement('div'); // Urgency logic const days = parseInt(item.daysLeft); const colorClass = days < 10 ? 'red' : days < 30 ? 'amber' : 'green'; card.className = `result-card ${colorClass}`; card.innerHTML = `
${item.title}
${item.description}
${item.deadline} ${item.tags.map(t => `${t}`).join('')} Apply ↗
`; list.innerHTML += card.outerHTML; }); chat.scrollTop = chat.scrollHeight; } catch (err) { chat.innerHTML += `
Error: Could not connect to the finder engine.
`; } finally { loading = false; document.getElementById('status-dot').className = 'sdot'; document.getElementById('status-text').innerText = 'ready'; } }
